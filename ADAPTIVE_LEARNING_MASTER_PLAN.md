### 2. Content Scoring Algorithm

```javascript
function scoreContent(content, userProfile) {
  // 1. Analyze content vocabulary
  const contentWords = extractWords(content.text);
  const knownWords = contentWords.filter(w => 
    userProfile.vocabulary.known.includes(w)
  );
  const unknownWords = contentWords.filter(w => 
    !userProfile.vocabulary.known.includes(w)
  );
  
  // 2. Calculate comprehension rate
  const comprehension = knownWords.length / contentWords.length;
  
  // 3. Score factors
  const scores = {
    // 40% - Comprehension (optimal: 85-95%)
    comprehension: scoreComprehension(comprehension),
    
    // 25% - New word count (optimal: 3-8 new words)
    newWords: scoreNewWords(unknownWords.length),
    
    // 20% - Interest match
    interest: scoreInterest(content.topics, userProfile.interests),
    
    // 10% - Level match
    level: scoreLevelMatch(content.level, userProfile.level),
    
    // 5% - Reinforcement (uses currently learning words)
    reinforcement: scoreReinforcement(
      contentWords, 
      userProfile.vocabulary.learning
    )
  };
  
  // 4. Calculate final score (0-100)
  const finalScore = 
    scores.comprehension * 0.40 +
    scores.newWords * 0.25 +
    scores.interest * 0.20 +
    scores.level * 0.10 +
    scores.reinforcement * 0.05;
  
  return {
    score: finalScore,
    comprehension: comprehension * 100,
    newWords: unknownWords,
    reasoning: generateExplanation(scores)
  };
}

function scoreComprehension(rate) {
  // Optimal zone: 85-95%
  if (rate >= 0.85 && rate <= 0.95) return 100;
  if (rate < 0.85) return rate * 100; // Too hard
  return Math.max(0, 100 - (rate - 0.95) * 200); // Too easy
}
```
