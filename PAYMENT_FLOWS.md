### Example: Unlimited Videos

```javascript
// Frontend
async function loadMoreVideos() {
    const response = await fetch('/api/videos/feed', {
        headers: {
            'Authorization': `Bearer ${token}`
        }
    });
    
    if (response.status === 403) {
        // Show upgrade modal
        showPremiumUpgradeModal();
        return;
    }
    
    const videos = await response.json();
    renderVideos(videos);
}

// Backend (server.js)
app.get('/api/videos/feed', requireAuth, async (req, res) => {
    const userId = req.user.id;
    const isPremium = checkPremiumStatus(userId);
    
    if (!isPremium) {
        // Free tier: limit to 5 videos
        const dailyCount = await getDailyVideoCount(userId);
        if (dailyCount >= 5) {
            return res.status(403).json({
                success: false,
                error: 'Daily video limit reached',
                upgradeUrl: '/premium.html'
            });
        }
    }
    
    // Premium or within limit: return videos
    const videos = await getVideos(userId, isPremium);
    res.json({ success: true, videos });
});
```
